1function [metal_stat,flag_stat]= checkMetal(cvVal,ref_cv,cvVal_all,avg_all)
 2% It checks the CV value and decides the metal presence.
 3% INPUT cvVal -> calculated cv value
 4%       ref_cv -> reference cv value for no metal case
 5%       variation_cv -> used for acceptable variation for tolerance
 6% OUTPUT metal_stat -> 1 BOX present and 0 Noise present
 7
 8%     max_cv=ref_cv-variation_cv; %#codegen
 9    flag_initial=0;
10    flag_last=0;
11
12    if cvVal>=ref_cv && avg_all(2)> avg_all(1) 
13        metal_stat=1; %BOX present
14    %     disp('CONTAMINATED BOX--METAL DETECTED');
15    else
16        metal_stat=0; 
17    %    disp('CLEAN BOX-- NO METAL PRESENT');
18    end
19    
20    if cvVal_all(1)>cvVal_all(2) || avg_all(1)>avg_all(2)
21        flag_initial=1;
22    elseif avg_all(3)>avg_all(2)
23        flag_last=1;
24    end
25    flag_stat =[flag_initial,flag_last];
26end
